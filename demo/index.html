Usage Examples

Basic Usage

import { SeoToast } from 'seo-toast';
import 'seo-toast/dist/style.css';

// Show different types of toasts
SeoToast.success('Operation completed successfully!');
SeoToast.error('Something went wrong');
SeoToast.warning('Please check your input');
SeoToast.info('New update available');

Position Control

// Configure toast position
SeoToast.getInstance({ 
  // top-left, top-center, bottom-right, etc.
  position: 'top-right' 
});

SeoToast.success('Positioned toast!');

Animation Types

// Different animation types
SeoToast.getInstance({
  enterAnimation: 'bounce',
  exitAnimation: 'fade'
});

SeoToast.info('Animated toast!');

Custom Icons

// Using custom image
SeoToast.success('Profile updated!', {
  customIcon: '/profile-icon.png'
});

// Using custom SVG
const customSvg = `<svg viewBox="0 0 24 24">...</svg>`;
SeoToast.info('Custom icon', { 
  customIcon: customSvg 
});

React Integration

import { SeoToast } from 'seo-toast';

function MyComponent() {
  const handleSubmit = async () => {
    try {
      await submitForm();
      SeoToast.success('Form submitted!');
    } catch (error) {
      SeoToast.error('Submission failed');
    }
  };

  return <button onClick={handleSubmit}>Submit</button>;
}

Vue Integration

// Vue 3 Composition API
import { SeoToast } from 'seo-toast';

export default {
  setup() {
    const handleClick = () => {
      SeoToast.success('Vue toast!');
    };

    return { handleClick };
  }
};

Advanced Configuration

// Custom timing and events
SeoToast.getInstance().addEventListener('toast-close', (event) => {
  console.log('Toast closed:', event.detail);
});

SeoToast.warning('Custom timing', {
  closeTime: 5000
});

API Reference

Static Methods

SeoToast.success(message, options?)

Shows a success toast

SeoToast.error(message, options?)

Shows an error toast

SeoToast.warning(message, options?)

Shows a warning toast

SeoToast.info(message, options?)

Shows an info toast

Configuration

position

'top-right' | 'top-left' | 'bottom-right' | 'bottom-left' | 'top-center' | 'bottom-center'

enterAnimation

'slide' | 'fade' | 'scale' | 'bounce'

exitAnimation

'slide' | 'fade' | 'scale' | 'bounce'

Options

closeTime?: number

Auto-close timeout in milliseconds

customIcon?: string

Custom icon URL or SVG string
